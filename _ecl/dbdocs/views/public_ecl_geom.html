ecl_geom(View)
Nameecl_geom
SQL 
SELECT
    DISTINCT dfl.submission_id,
    dfg.dataset_file_list_id,
    dfg.geometry AS geom
FROM dataset_file_geometry dfg
    JOIN dataset_file_list dfl ON dfl.dataset_file_list_id = dfg.dataset_file_list_id
WHERE dfg.geometry IS NOT NULL
