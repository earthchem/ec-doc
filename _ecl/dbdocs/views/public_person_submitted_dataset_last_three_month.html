person_submitted_dataset_last_three_month(View)
Nameperson_submitted_dataset_last_three_month
SQL 
SELECT
    DISTINCT g.first_name,
    g.last_name,
    g.email
FROM submissions s
    JOIN grl_user g ON g.email::text <> 'info@earthchem.org'::text
        AND g.account_id = s.owner
        AND g.last_name::text <> 'EarthChem'::text
        AND (s.status_id = ANY (ARRAY[2,
    1,
    0]))
        AND s.created_date > '2019-11-30 00:00:00'::timestamp without time zone
ORDER BY g.last_name,
    g.first_name,
    g.email
